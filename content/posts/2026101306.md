### 第三步：获取实测代码 - 双均线策略示例

```python
from binance.client import Client
from binance.enums import *
import pandas as pd
import ta  # 技术指标库
import time
from config import API_KEY, API_SECRET

# 初始化客户端，连接到U本位合约测试网（正式交易请移除`testnet=True`）
client = Client(API_KEY, API_SECRET, testnet=True)  # 强烈建议先在测试网运行！

def get_klines(symbol='BTCUSDT', interval='15m', limit=100):
    """获取K线数据"""
    klines = client.futures_klines(symbol=symbol, interval=interval, limit=limit)
    df = pd.DataFrame(klines, columns=['timestamp', 'open', 'high', 'low', 'close', 'volume', 'close_time', 'quote_asset_volume', 'number_of_trades', 'taker_buy_base_asset_volume', 'taker_buy_quote_asset_volume', 'ignore'])
    df['close'] = df['close'].astype(float)
    return df

def calculate_indicators(df):
    """计算技术指标：快慢均线"""
    df['ma_fast'] = ta.trend.sma_indicator(df['close'], window=10)
    df['ma_slow'] = ta.trend.sma_indicator(df['close'], window=30)
    return df

def check_crossover(df):
    """检查金叉死叉信号"""
    last_row = df.iloc[-1]
    prev_row = df.iloc[-2]
    
    # 金叉：快线上穿慢线
    if prev_row['ma_fast'] <= prev_row['ma_slow'] and last_row['ma_fast'] > last_row['ma_slow']:
        return 'BUY'
    # 死叉：快线下穿慢线
    elif prev_row['ma_fast'] >= prev_row['ma_slow'] and last_row['ma_fast'] < last_row['ma_slow']:
        return 'SELL'
    return 'HOLD'

def place_order(symbol, side, quantity=0.001):
    """下达合约订单（测试网）"""
    try:
        order = client.futures_create_order(
            symbol=symbol,
            side=side,
            type='MARKET',  # 市价单
            quantity=quantity
        )
        print(f"订单已下达: {order}")
        return order
    except Exception as e:
        print(f"下单失败: {e}")
        return None

# 主策略循环（示例，实际应用需加入风控和日志）
symbol = 'BTCUSDT'
while True:
    df = get_klines(symbol)
    df = calculate_indicators(df)
    signal = check_crossover(df)
    
    if signal == 'BUY':
        print(f"{time.ctime()} - 检测到金叉信号，执行买入")
        place_order(symbol, 'BUY')
    elif signal == 'SELL':
        print(f"{time.ctime()} - 检测到死叉信号，执行卖出")
        place_order(symbol, 'SELL')
    else:
        print(f"{time.ctime()} - 无信号，持仓观望")
    
    time.sleep(60)  # 每分钟检查一次
```
