1.  在项目目录下创建`.env`文件，存放你的密钥：

```python
import ccxt
import pandas as pd
from datetime import datetime
import os
from dotenv import load_dotenv

# 加载环境变量
load_dotenv()

# 初始化币安合约交易所对象（使用测试网请将`enableRateLimit`设为True，并使用testnet参数）
exchange = ccxt.binance({
    'apiKey': os.getenv('BINANCE_API_KEY'),
    'secret': os.getenv('BINANCE_SECRET_KEY'),
    'options': {
        'defaultType': 'future',  # 指定为合约交易
    },
    'enableRateLimit': True,  # 遵守速率限制，必须为True
})

def check_connection():
    """测试连接并获取账户余额（USDT合约账户）"""
    try:
        # 获取合约账户余额
        balance = exchange.fetch_balance()
        usdt_balance = balance['USDT']['total']
        print(f"[{datetime.now().isoformat()}] 连接成功！合约账户USDT余额: {usdt_balance}")
        return True
    except Exception as e:
        print(f"[{datetime.now().isoformat()}] 连接失败: {e}")
        return False

if __name__ == "__main__":
    check_connection()
```
